Study form html5 canvas 详细使用教程

  • 创建上下文

    var canvas = document.getElementById("canvas"); var context = canvas.getContext("2d");

  • 绘制图像

    context.fill(); // 填充
    context.stroke(); // 绘制边框

  • 绘制的样式

    context.fillStyle // 填充的样式
    context.strokeStyle // 边框的样式
    context.lineWidth // 边框宽度

  • 颜色的表示方式

    与css一致

  • api

    绘制矩形

    context.fillRect(x, y, width, height);
    context.strokeRect(x, y, width, height);

    清除矩形区域

    context.clearRect(x, y, width, height);

    圆弧 ( )

    context.arc(x, y, radius, startAngle, endAngle, anticlockwise);
    // x,y 圆心横纵坐标
    // anticlockwise: 逆时针： true, 顺时针： false

    线段

    context.moveTo(x, y); // 定义线条开始坐标
    context.lineTo(x, y); // 定义线条结束坐标

    曲线

    context.bezierCurveTo(cp1x, cp1y, cp2x, cp2y, x, y); //贝塞尔曲线
    context.quadraticCurveTo(qcpx, qapy, qx, qy);

    渐变


    var lg = context.createLinearGradient(xStart, yStart, xEnd, yEnd);
    lg.addColorStop(offset, color);
    // offset: 设置颜色离渐变结束点的偏移量（0~1）

    var rg = content.createRadialGrandient(xStart, yStart, radiusStart, xEnd, yEnd, radiusEnd);
    // xStart, yStart,: 发散开始圆心； xEnd, yEnd： 发散结束圆心; radiusStart, radiusEnd: 发散开始半径, 发散结束半径
    rg.addColorStrop(offset, color);

    绘制文本

    context.fillText(text, x, y);
    content.strokeText(text, x, y);

    图形变形


    context.translate(x, y);
    // x, y：平移的距离

    context.scale(x, y);

    context.rotate(angle)

    矩阵变换

    context.transform(m11, m12, m21, m22, dx, dy)
